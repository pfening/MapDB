public class dictionaryBean {
int wordid;
int userid;
int langid;
String m_word;
String f_word;
public dictionaryBean() {}
public dictionaryBean(int wordid, int userid, int langid, String m_word, String f_word) {
this.wordid = wordid
this.userid = userid
this.langid = langid
this.m_word = m_word
this.f_word = f_word
}
}
